Random Number Generator in Python

This script helps you create random numbers easily. Just set the number of iterations and the range, and it does the rest.

You will need Python 3.8.0 or higher and the "random" library.

import random

num_iteraciones = None
num_range = None

def generar_numeros_aleatorios(num_iteraciones, num_range):
    for _ in range(num_iteraciones):
        numero_aleatorio = random.randint(num_range[0], num_range[1])
        print(numero_aleatorio)

def def_num_iteraciones():
    global num_iteraciones, num_range
    if num_iteraciones is None or num_iteraciones > 2**16 or num_iteraciones < 2**16 * -1:
        num_iteraciones = int(input("Number of iterations: "))
        def_num_iteraciones()
    elif num_range is None or num_range > [2**16, 2**16] or num_range < [2**16 * -1, 2**16 * -1]:
        num_range = [int(input("Minimum value: ")), int(input("Maximum value: "))]
        def_num_iteraciones()
    else:
        if num_range[0] > num_range[1]:
            num_range = [num_range[1], num_range[0]]
        generar_numeros_aleatorios(num_iteraciones, num_range)

def_num_iteraciones()

input('Press "Enter" to exit.')
